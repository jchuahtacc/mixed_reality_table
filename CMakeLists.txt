cmake_minimum_required(VERSION 3.1)
set(CMAKE_CXX_STANDARD 11)
project(mixed_reality_table)
include_directories(lib)
find_package( OpenCV REQUIRED )
add_executable(generate_markers src/generate_markers.cpp)
add_executable(generate_charuco src/generate_charuco.cpp)
add_executable(detect_markers src/detect_markers.cpp)
add_executable(calibrate_camera src/calibrate_camera.cpp)
add_executable(detect_blobs src/detect_blobs.cpp)
add_executable(detect_contours src/detect_contours.cpp)
add_executable(detect_server src/detect_server.cpp)
target_link_libraries(generate_markers ${OpenCV_LIBS})
target_link_libraries(detect_markers ${OpenCV_LIBS})
target_link_libraries(calibrate_camera ${OpenCV_LIBS})
target_link_libraries(generate_charuco ${OpenCV_LIBS})
target_link_libraries(detect_blobs ${OpenCV_LIBS})
target_link_libraries(detect_contours ${OpenCV_LIBS})
target_link_libraries(detect_server ${OpenCV_LIBS})
